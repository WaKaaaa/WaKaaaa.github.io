Java算法学习--回溯法--n后问题

  • 回溯法 n后问题
  view

一：问题描述

在nn格的棋盘上放着彼此不受攻击的n个皇后，按照国际象棋规定，皇后可以攻击与之处在同一行，同一列，对角线上的皇后。n后问题等价于：在nn的棋盘上，任意两个皇后不在同一行，同一列，同一斜线，当n的值一定时（比如8），有多少种摆放的方法。

算法：

1.递归回溯：

package suanfasheji;

import java.util.Scanner;

public class NQueen1 {
    static int n; // 皇后个数
    static int[] x; // 当前解
    static long sum; //当前已经找到的可行解的方案个数

    public static long nQueen(int nn){
        n = nn;
        sum = 0;
        x = new int[n+1];
        for(int i=0; i<=n; i++) x[i]=0;
        backtrack(1);    //从第一层开始搜索
        return sum;
    }

    private static boolean palce(int k){
        for(int j=1; j<k; j++)
            if(Math.abs(k-j) == Math.abs(x[j]-x[k]) || (x[j]==x[k])) return false;
        return true;
    }

    private static void backtrack(int t){
        if(t>n) sum++;
        else
            for(int i=1; i<=n; i++){
                x[t] = i;
                if(palce(t)) backtrack(t+1);
            }
    }
    public static void main(String[] args) {
        // TODO Auto-generated method stub
        Scanner in = new Scanner(System.in);
        System.out.println("请输入皇后数：");
        int n = in.nextInt();
        NQueen1 qun = new NQueen1();
        long sum = qun.nQueen(n);
        System.out.print(n+"皇后的可行解是："+sum);
    }

}

2.非递归回溯：

package suanfasheji;

import java.util.Scanner;

public class NQueen2 {
    static int n; // 皇后个数
    static int[] x; // 当前解
    static long sum; // 当前已经找到的可行解的方案个数

    public static long nQueen(int nn) {
        n = nn;
        sum = 0;
        x = new int[n + 1];
        for (int i = 0; i <= n; i++)
            x[i] = 0;
        backtrack(); 
        return sum;
    }

    private static boolean palce(int k) {
        for (int j = 1; j < k; j++)
            if (Math.abs(k - j) == Math.abs(x[j] - x[k]) || (x[j] == x[k]))
                return false;
        return true;
    }

    private static void backtrack(){
        x[1] = 0;
        int k=1;
        while(k>0){
            x[k]+=1;
            while((x[k]<=n) && !(palce(k))) x[k]+=1;
            if(x[k]<=n)
                if(k==n) sum++;
                else{
                    k++;
                    x[k]=0;
                }
            else k--;
        }
    }

    public static void main(String[] args) {
        // TODO Auto-generated method stub
        Scanner in = new Scanner(System.in);
        System.out.println("请输入皇后数：");
        int n = in.nextInt();
        NQueen1 qun = new NQueen1();
        long sum = qun.nQueen(n);
        System.out.print(n+"皇后的可行解是："+sum);
    }

}
