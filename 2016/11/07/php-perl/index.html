php正则表达式


<?php 

// 如果整个规则模式匹配通过而被匹配的字符串里还有字符，也是可以匹配成功的

// 只要模式规则全部通过，就可以匹配成功
// 此实例中，只要有php三个字符相连出现，模式匹配就成功
// 但是匹配和 == 是不相同的，匹配可以使三个字符对多个字符

// $pattern = '/php/';    // 规则模式
// $subject = 'gggphpqqq';    // 被匹配的字符串


// 前导的概念：就是他前面的一个字符
// + * ？ {} . $ ^ |


// 1. h+ 表示至少包含一个前导字符，也就是h大于等于1个
// $pattern = '/ph+p/';    // 规则模式


// 2. h*表示包含0到无数个前导字符，也就是h的个数可以为任意个
// $pattern = '/ph?p/';    // 规则模式

// 3. h?表示包含0个或者1个前导字符，也就是h的个数等于0个或者1个


// 4. h.表示前导字符可以为任意字符，一个.是一个字符，将.号代替一个字符
// 5. .* 表示前导任意多个任意字符，0到无数个都可以
// $pattern = '/ph.p/';    // 规则模式
// $subject = 'phjp';    // 被匹配的字符串（匹配）


// 5. $号加在字符串的末尾，只要末尾的几个字符匹配即可

// 6. 相对的是^，通常在字符串的开始，只要开始的几个字符匹配即可

// 7. | 是条件选择符号，如果其中有一个符合就返回1
// $pattern = '/php|jsp|asp/';    // 规则模式
// $subject = 'phpe';    // 被匹配的字符串(匹配)，如果是这三个字符串之外的就不符合

// 8. h{3} 表示前导有三个，不能多也不能少
// h{3,5} 表示前导有3个到5都可以，多了少了都不行
// h{5,} 如果后边没有数字，就表示从5个到大于5个的人一多个前导字符
$pattern = '/ph{3}p/';    // 规则模式
$subject = 'phhhpe';    // 被匹配的字符串

if(preg_match($pattern, $subject)){
    echo "匹配";
}else{
    echo "不匹配";
}

?>
