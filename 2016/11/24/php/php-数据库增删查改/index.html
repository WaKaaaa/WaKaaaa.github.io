php学习--数据库增删查改

  view

一： 新增数据

// 注意：这里username是primary key，只能增加一次，所以重复操作会显示错误信息
$query = 'INSERT INTO userdetail(username,userpass,regtime) VALUES("fas ",123456,NOW())';
@mysql_query($query) or die("新增数据错误".mysql_error());

二： 修改数据

$query = 'UPDATE userdetail SET userpass = 897 WHERE username="张三"';
@mysql_query($query) or die("修改数据失败".mysql_error());

三： 删除数据

$query = 'DELETE FROM userdetail WHERE username="张三"';
@mysql_query($query) or die("删除数据失败".mysql_error());

四： 显示数据

$query = "SELECT username, userpass FROM userdetail";
$result = mysql_query($query) or die("sql语句有误".mysql_error());
while(!!$row = mysql_fetch_array($result)){
    echo $row['username']."-----".$row["userpass"];
    echo "<br>";
}
